<!-- typeof 연산자 -->
> typeof 'hello world'; // 자료형 확인
< 'string'

<!-- 템플릿 리터럴 -->
<!-- 백틱(``, backtick)으로 감싸는 방법 -->

> `hello world`;

<!-- 지수표기법 -->
<!-- e 뒤에 나오는 숫자만큼 10의 거듭제곱 -->
> 5e4; // 5 * 10000 = 50000
> 5e+4; // 5 * 10000 = 50000
> 5e-3; // 5 * (1/1000) = 0.005

<!-- 2, 8, 16 진법 표기 -->
<!-- 2진법은 숫자 앞에 0b(숫자 0, 영어 b)를 붙임 -->
> 0b11;
< 3

<!-- 8진법은 숫자 앞에 0o(숫자 0, 영어 o)를 붙임 -->
> 0o15;
< 13

<!-- 16진법은 숫자 앞에 0x(숫자 0, 영어 x)를 붙임 -->
> 0x1c;
< 28

<!-- 문자열 -> 숫자로 변경 -->
> typeof parseInt('5');
< 'number'

> parseInt('3.14');
< 3
> parseFloat('3.14');
< 3.14
